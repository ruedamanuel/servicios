MySQL Database Service

Manages the connection pool that interfaces with the MySQL database that manages the persistent storage of the application.

Initializes MySQL connection pool.

Method Mysql.query

Checks out a connection and performs the query provided.

Parameters:

  • query must be a String.

  • values must be an Array.

Returns a Promise

Example:

mysql.query("SELECT * FROM profession WHERE id > ?", [5])
  .then(function(result){
    log.info("it works: ", result);
  })
  .catch(function(e){
    log.error("Failed: ", e);
  });
query: function(query, values){ return q.Promise(function(resolve, reject){ pool.getConnection(function(error, connection){ if(error){ reject(error); }else{ connection.query(query, values, function(err, rows){ if(err){ reject(err); }else{ resolve(format.camelCaseArray(rows)); } connection.release(); }); } }); }); } };
